module.exports = {
  // ...
  webpackFinal: (config) => {
    config.module.rules.push({
      test: /\.(scss)$/,
      use: [
        "style-loader",
        "css-loader",
        "sass-loader",
        {
          loader: require.resolve("sass-resources-loader"),
          options: {
            resources: `<path>/global.scss`,
          },
        },
      ],
      include: path.resolve(<path>),
    });

    return config;
  },
};
